chi4 = np.zeros((N))
def Chi2Analysis(i):
    idx = np.zeros((N+1,N))
    for j in range(N):
        addi = []
        for k in range(0,307,1):
            d = integrate.quad(lambda z: 1/(H0*np.sqrt(Om[i]*(1/(1+z))**(-3) + Ol[j])),0,r[k])
            f = 5*np.log10(3*10**5*(1+r[k])*np.abs(d[0]))+25
            diff = f - ob_data[k]
            d2 = (diff**2/sd[k]**2)
            addi.append(d2)
        c = np.sum(addi)
        chi4[j] = c
        #print(i,j)
    return chi4

def marginal_PDF(x,pdf,xl,yl,t,fn):
    plt.style.use('classic')
    plt.style.use('seaborn-white')
    mean = sum(x *pdf) / sum(pdf)
    sigma = np.sqrt(sum(pdf * (x - mean) ** 2) / sum(pdf))
    y = norm.pdf(x, mean, sigma)
    y2 = y/sum(y)
    #plt.figure(figsize=(7,7))
    
    plt.plot(x, y2,'r-',label="Normal fit\n$\mu$ = %.4f" %mean+ "\n$\sigma$ = %.4f" %sigma)
    plt.scatter(x,pdf,label="%d" %len(r)+" Data")
    plt.legend()
    plt.xlim(min(x),max(x))
    plt.ylim(min(pdf)-max(pdf)/25,max(pdf)+max(pdf)/25)
    plt.minorticks_on()
    plt.tick_params(which='major',direction='in',length=7,labeltop=False,
                    labelleft=True,labelright=False,width=2,labelsize=20)
    plt.tick_params(which='minor',direction='in',length=4,width=2)
    plt.xlabel(xl,fontsize=20,fontweight='bold')
    plt.ylabel(yl,fontsize=20,fontweight='bold')
    plt.title(t+"\n"+xl+"= %.3f" %mean+" $\pm$ %.3f"%sigma,fontsize=15,fontweight='bold',loc='right',y=1.01,alpha=0.6)
    plt.savefig(fn,bbox_inches = 'tight')
